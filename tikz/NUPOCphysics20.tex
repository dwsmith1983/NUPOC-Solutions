\newcommand{\bul}[1]{
  \begin{scope}[#1]
    \fill[draw = black!80, top color = black!10, bottom color = black!70]
    (0, 1mm) -- (1.5mm, 1mm) to [out = 0, in = 120] (2.5mm, 0.5mm) 
    to[out = -120, in = 0] (1.5mm, 0) -- (0, 0) -- cycle;
    \draw[white, draw opacity = 0.5] (0, 0.1mm) -- (2.3mm, 0.1mm);
  \end{scope}
}
